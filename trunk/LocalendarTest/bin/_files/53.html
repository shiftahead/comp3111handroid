451    private String validNameStringFromInteger(long i) {
452        String s = Long.toString(i);
453        StringBuilder result = new StringBuilder("Perm");
454 
455        // We know each character is a digit. Convert it into a letter 'a'-'j'. Avoid repeated characters
456        //  that might make Facebook reject the name by converting every other repeated character into one
457        //  10 higher ('k'-'t').
458        char lastChar = 0;
459        for (char c : s.toCharArray()) {
460            if (c == lastChar) {
461                c += 10;
462            }
463            result.append((char) (c + 'a' - '0'));
464            lastChar = c;
465        }
466 
467        return result.toString();
468    }
