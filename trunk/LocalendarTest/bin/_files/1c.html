714    void updateAdapter(SimpleGraphObjectCursor<T> data) {
715        if (adapter != null) {
716            // As we fetch additional results and add them to the table, we do not
717            // want the items displayed jumping around seemingly at random, frustrating the user's
718            // attempts at scrolling, etc. Since results may be added anywhere in
719            // the table, we choose to try to keep the first visible row in a fixed
720            // position (from the user's perspective). We try to keep it positioned at
721            // the same offset from the top of the screen so adding new items seems
722            // smoother, as opposed to having it "snap" to a multiple of row height
723 
724            // We use the second row, to give context above and below it and avoid
725            // cases where the first row is only barely visible, thus providing little context.
726            // The exception is where the very first row is visible, in which case we use that.
727            View view = listView.getChildAt(1);
728            int anchorPosition = listView.getFirstVisiblePosition();
729            if (anchorPosition > 0) {
730                anchorPosition++;
731            }
732            GraphObjectAdapter.SectionAndItem<T> anchorItem = adapter.getSectionAndItem(anchorPosition);
733            final int top = (view != null &&
734                    anchorItem.getType() != GraphObjectAdapter.SectionAndItem.Type.ACTIVITY_CIRCLE) ?
735                    view.getTop() : 0;
736 
737            // Now actually add the results.
738            boolean dataChanged = adapter.changeCursor(data);
739 
740            if (view != null && anchorItem != null) {
741                // Put the item back in the same spot it was.
742                final int newPositionOfItem = adapter.getPosition(anchorItem.sectionKey, anchorItem.graphObject);
743                if (newPositionOfItem != -1) {
744                    listView.setSelectionFromTop(newPositionOfItem, top);
745                }
746            }
747 
748            if (dataChanged && onDataChangedListener != null) {
749                onDataChangedListener.onDataChanged(PickerFragment.this);
750            }
751        }
752    }
