653            Context context = getContext();
654            final Session openSession = sessionTracker.getOpenSession();
655 
656            if (openSession != null) {
657                // If the Session is currently open, it must mean we need to log out
658                if (confirmLogout) {
659                    // Create a confirmation dialog
660                    String logout = getResources().getString(R.string.com_facebook_loginview_log_out_action);
661                    String cancel = getResources().getString(R.string.com_facebook_loginview_cancel_action);
662                    String message;
663                    if (user != null && user.getName() != null) {
664                        message = String.format(getResources().getString(R.string.com_facebook_loginview_logged_in_as), user.getName());
665                    } else {
666                        message = getResources().getString(R.string.com_facebook_loginview_logged_in_using_facebook);
667                    }
668                    AlertDialog.Builder builder = new AlertDialog.Builder(context);
669                    builder.setMessage(message)
670                           .setCancelable(true)
671                           .setPositiveButton(logout, new DialogInterface.OnClickListener() {
672                               public void onClick(DialogInterface dialog, int which) {
673                                   openSession.closeAndClearTokenInformation();
674                               }
675                           })
676                           .setNegativeButton(cancel, null);
677                    builder.create().show();
678                } else {
679                    openSession.closeAndClearTokenInformation();
680                }
681            } else {
682                Session currentSession = sessionTracker.getSession();
683                if (currentSession == null || currentSession.getState().isClosed()) {
684                    sessionTracker.setSession(null);
685                    Session session = new Session.Builder(context).setApplicationId(applicationId).build();
686                    Session.setActiveSession(session);
687                    currentSession = session;
688                }
689                if (!currentSession.isOpened()) {
690                    Session.OpenRequest openRequest = null;
691                    if (parentFragment != null) {
692                        openRequest = new Session.OpenRequest(parentFragment);
693                    } else if (context instanceof Activity) {
694                        openRequest = new Session.OpenRequest((Activity)context);
695                    }
696 
697                    if (openRequest != null) {
698                        openRequest.setDefaultAudience(properties.defaultAudience);
699                        openRequest.setPermissions(properties.permissions);
700                        openRequest.setLoginBehavior(properties.loginBehavior);
701 
702                        if (SessionAuthorizationType.PUBLISH.equals(properties.authorizationType)) {
703                            currentSession.openForPublish(openRequest);
704                        } else {
705                            currentSession.openForRead(openRequest);
706                        }
707                    }
708                }
709            }
710 
711            AppEventsLogger logger = AppEventsLogger.newLogger(getContext());
712 
713            Bundle parameters = new Bundle();
714            parameters.putInt("logging_in", (openSession != null) ? 0 : 1);
715 
716            logger.logSdkEvent(loginLogoutEventName, null, parameters);
717        }
