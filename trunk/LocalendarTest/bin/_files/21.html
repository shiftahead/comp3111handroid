273     * @param maxPaddingSize the size at which to apply maximum padding for the dialog
274     * @return a scaled size.
275     */
276    private int getScaledSize(int screenSize, float density, int noPaddingSize, int maxPaddingSize) {
277        int scaledSize = (int) ((float) screenSize / density);
278        double scaleFactor;
279        if (scaledSize <= noPaddingSize) {
280            scaleFactor = 1.0;
281        } else if (scaledSize >= maxPaddingSize) {
282            scaleFactor = MIN_SCALE_FACTOR;
283        } else {
284            // between the noPadding and maxPadding widths, we take a linear reduction to go from 100%
285            // of screen size down to MIN_SCALE_FACTOR
286            scaleFactor = MIN_SCALE_FACTOR +
287                    ((double) (maxPaddingSize - scaledSize))
288                            / ((double) (maxPaddingSize - noPaddingSize))
289                            * (1.0 - MIN_SCALE_FACTOR);
290        }
291        return (int) (screenSize * scaleFactor);
292    }

360        public boolean shouldOverrideUrlLoading(WebView view, String url) {
361            Utility.logd(LOG_TAG, "Redirect URL: " + url);
362            if (url.startsWith(WebDialog.REDIRECT_URI)) {
363                Bundle values = Util.parseUrl(url);
364 
365                String error = values.getString("error");
366                if (error == null) {
367                    error = values.getString("error_type");
368                }
369 
370                String errorMessage = values.getString("error_msg");
371                if (errorMessage == null) {
372                    errorMessage = values.getString("error_description");
373                }
374                String errorCodeString = values.getString("error_code");
375                int errorCode = FacebookRequestError.INVALID_ERROR_CODE;
376                if (!Utility.isNullOrEmpty(errorCodeString)) {
377                    try {
378                        errorCode = Integer.parseInt(errorCodeString);
379                    } catch (NumberFormatException ex) {
380                        errorCode = FacebookRequestError.INVALID_ERROR_CODE;
381                    }
382                }
383 
384                if (Utility.isNullOrEmpty(error) && Utility
385                        .isNullOrEmpty(errorMessage) && errorCode == FacebookRequestError.INVALID_ERROR_CODE) {

388                        error.equals("OAuthAccessDeniedException"))) {

391                    FacebookRequestError requestError = new FacebookRequestError(errorCode, error, errorMessage);
392                    sendErrorToListener(new FacebookServiceException(requestError, errorMessage));
393                }
394 
395                WebDialog.this.dismiss();

397            } else if (url.startsWith(WebDialog.CANCEL_URI)) {

400                return true;
401            } else if (url.contains(DISPLAY_TOUCH)) {
402                return false;
403            }
404            // launch non-dialog URLs in a full browser
