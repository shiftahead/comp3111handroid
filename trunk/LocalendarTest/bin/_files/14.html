451    public static List<LatLng> decode(final String encodedPath) {
452        int len = encodedPath.length();
453 
454        // For speed we preallocate to an upper bound on the final length, then
455        // truncate the array before returning.
456        final List<LatLng> path = new ArrayList<LatLng>();
457        int index = 0;
458        int lat = 0;
459        int lng = 0;
460 
461        while (index < len) {
462            int result = 1;
463            int shift = 0;
464            int b;
465            do {
466                b = encodedPath.charAt(index++) - 63 - 1;
467                result += b << shift;
468                shift += 5;
469            } while (b >= 0x1f);
470            lat += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
471 
472            result = 1;
473            shift = 0;
474            do {
475                b = encodedPath.charAt(index++) - 63 - 1;
476                result += b << shift;
477                shift += 5;
478            } while (b >= 0x1f);
479            lng += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
480 
481            path.add(new LatLng(lat * 1e-5, lng * 1e-5));
482        }
483 
484        return path;
485    }
