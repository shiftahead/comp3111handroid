843    private static void handleResponse(AccessTokenAppIdPair accessTokenAppId, Request request, Response response,
844            SessionEventsState sessionEventsState, FlushStatistics flushState) {
845        FacebookRequestError error = response.getError();
846        String resultDescription = "Success";
847 
848        FlushResult flushResult = FlushResult.SUCCESS;
849 
850        if (error != null) {
851            final int NO_CONNECTIVITY_ERROR_CODE = -1;
852            if (error.getErrorCode() == NO_CONNECTIVITY_ERROR_CODE) {
853                resultDescription = "Failed: No Connectivity";
854                flushResult = FlushResult.NO_CONNECTIVITY;
855            } else {
856                resultDescription = String.format("Failed:\n  Response: %s\n  Error %s",
857                        response.toString(),
858                        error.toString());
859                flushResult = FlushResult.SERVER_ERROR;
860            }
861        }
862 
863        if (Settings.isLoggingBehaviorEnabled(LoggingBehavior.APP_EVENTS)) {
864            String eventsJsonString = (String) request.getTag();
865            String prettyPrintedEvents;
866 
867            try {
868                JSONArray jsonArray = new JSONArray(eventsJsonString);
869                prettyPrintedEvents = jsonArray.toString(2);
870            } catch (JSONException exc) {
871                prettyPrintedEvents = "<Can't encode events for debug logging>";
872            }
873 
874            Logger.log(LoggingBehavior.APP_EVENTS, TAG,
875                    "Flush completed\nParams: %s\n  Result: %s\n  Events JSON: %s",
876                    request.getGraphObject().toString(),
877                    resultDescription,
878                    prettyPrintedEvents);
879        }
880 
881        sessionEventsState.clearInFlightAndStats(error != null);
882 
883        if (flushResult == FlushResult.NO_CONNECTIVITY) {
884            // We may call this for multiple requests in a batch, which is slightly inefficient since in principle
885            // we could call it once for all failed requests, but the impact is likely to be minimal.
886            // We don't call this for other server errors, because if an event failed because it was malformed, etc.,
887            // continually retrying it will cause subsequent events to not be logged either.
888            PersistedEvents.persistEvents(applicationContext, accessTokenAppId, sessionEventsState);
889        }
890 
891        if (flushResult != FlushResult.SUCCESS) {
892            // We assume that connectivity issues are more significant to report than server issues.
893            if (flushState.result != FlushResult.NO_CONNECTIVITY) {
894                flushState.result = flushResult;
895            }
896        }
897    }
