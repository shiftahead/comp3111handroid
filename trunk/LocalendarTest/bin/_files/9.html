558 
559        // For speed we preallocate to an upper bound on the final length, then
560        // truncate the array before returning.
561        final List<LatLng> path = new ArrayList<LatLng>();
562        int index = 0;
563        int lat = 0;
564        int lng = 0;
565 
566        while (index < len) {
567            int result = 1;
568            int shift = 0;
569            int b;
570            do {
571                b = encodedPath.charAt(index++) - 63 - 1;
572                result += b << shift;
573                shift += 5;
574            } while (b >= 0x1f);
575            lat += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
576 
577            result = 1;
578            shift = 0;
579            do {
580                b = encodedPath.charAt(index++) - 63 - 1;
581                result += b << shift;
582                shift += 5;
583            } while (b >= 0x1f);
584            lng += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
