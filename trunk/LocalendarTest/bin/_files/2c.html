500    private static int getLatestAvailableProtocolVersionForAppInfo(Context context, NativeAppInfo appInfo,
501            final int minimumVersion) {
502        ContentResolver contentResolver = context.getContentResolver();
503 
504        String [] projection = new String[]{ PLATFORM_PROVIDER_VERSION_COLUMN };
505        Uri uri = buildPlatformProviderVersionURI(appInfo);
506        Cursor c = contentResolver.query(uri, projection, null, null, null);
507        if (c == null) {
508            return NO_PROTOCOL_AVAILABLE;
509        }
510 
511        Set<Integer> versions = new HashSet<Integer>();
512        while (c.moveToNext()) {
513            int version = c.getInt(c.getColumnIndex(PLATFORM_PROVIDER_VERSION_COLUMN));
514            versions.add(version);
515        }
516 
517        for (Integer knownVersion : KNOWN_PROTOCOL_VERSIONS) {
518            if (knownVersion < minimumVersion) {
519                return NO_PROTOCOL_AVAILABLE;
520            }
521 
522            if (versions.contains(knownVersion)) {
523                return knownVersion;
524            }
525        }
526 
527        return NO_PROTOCOL_AVAILABLE;
528    }
