225    void tryNextHandler() {

362    RequestBatch createReauthValidationBatch(final Result pendingResult) {
363        // We need to ensure that the token we got represents the same fbid as the old one. We issue
364        // a "me" request using the current token, a "me" request using the new token, and a "me/permissions"
365        // request using the current token to get the permissions of the user.
366 
367        final ArrayList<String> fbids = new ArrayList<String>();
368        final ArrayList<String> tokenPermissions = new ArrayList<String>();
369        final String newToken = pendingResult.token.getToken();
370 
371        Request.Callback meCallback = new Request.Callback() {
372            @Override
373            public void onCompleted(Response response) {
374                try {
375                    GraphUser user = response.getGraphObjectAs(GraphUser.class);
376                    if (user != null) {
377                        fbids.add(user.getId());
378                    }
379                } catch (Exception ex) {
380                }
381            }
382        };
383 
384        String validateSameFbidAsToken = pendingRequest.getPreviousAccessToken();
385        Request requestCurrentTokenMe = createGetProfileIdRequest(validateSameFbidAsToken);
386        requestCurrentTokenMe.setCallback(meCallback);
387 
388        Request requestNewTokenMe = createGetProfileIdRequest(newToken);
389        requestNewTokenMe.setCallback(meCallback);
390 
391        Request requestCurrentTokenPermissions = createGetPermissionsRequest(validateSameFbidAsToken);
392        requestCurrentTokenPermissions.setCallback(new Request.Callback() {
393            @Override
394            public void onCompleted(Response response) {
395                try {
396                    GraphMultiResult result = response.getGraphObjectAs(GraphMultiResult.class);
397                    if (result != null) {
398                        GraphObjectList<GraphObject> data = result.getData();
399                        if (data != null && data.size() == 1) {
400                            GraphObject permissions = data.get(0);
401 
402                            // The keys are the permission names.
403                            tokenPermissions.addAll(permissions.asMap().keySet());
404                        }
405                    }
406                } catch (Exception ex) {
407                }
408            }
409        });
410 
411        RequestBatch batch = new RequestBatch(requestCurrentTokenMe, requestNewTokenMe,
412                requestCurrentTokenPermissions);
413        batch.setBatchApplicationId(pendingRequest.getApplicationId());
414        batch.addCallback(new RequestBatch.Callback() {
415            @Override
416            public void onBatchCompleted(RequestBatch batch) {
417                try {
418                    Result result = null;
419                    if (fbids.size() == 2 && fbids.get(0) != null && fbids.get(1) != null &&
420                            fbids.get(0).equals(fbids.get(1))) {
421                        // Modify the token to have the right permission set.
422                        AccessToken tokenWithPermissions = AccessToken
423                                .createFromTokenWithRefreshedPermissions(pendingResult.token,
424                                        tokenPermissions);
425                        result = Result.createTokenResult(pendingRequest, tokenWithPermissions);
426                    } else {
427                        result = Result
428                                .createErrorResult(pendingRequest, "User logged in as different Facebook user.", null);
429                    }
430                    complete(result);
431                } catch (Exception ex) {
432                    complete(Result.createErrorResult(pendingRequest, "Caught exception", ex.getMessage()));
433                } finally {
434                    notifyBackgroundProcessingStop();
435                }
436            }
437        });
438 
439        return batch;
440    }
