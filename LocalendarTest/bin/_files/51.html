408            boolean isFromCache, Object originalResult) throws JSONException {
409        if (object instanceof JSONObject) {
410            JSONObject jsonObject = (JSONObject) object;
411 
412            FacebookRequestError error =
413                    FacebookRequestError.checkResponseAndCreateError(jsonObject, originalResult, connection);
414            if (error != null) {
415                if (error.getErrorCode() == INVALID_SESSION_FACEBOOK_ERROR_CODE) {
416                    Session session = request.getSession();
417                    if (session != null) {
418                        session.closeAndClearTokenInformation();
419                    }
420                }
421                return new Response(request, connection, error);
422            }
423 
424            Object body = Utility.getStringPropertyAsJSON(jsonObject, BODY_KEY, NON_JSON_RESPONSE_PROPERTY);
425 
426            if (body instanceof JSONObject) {
427                GraphObject graphObject = GraphObject.Factory.create((JSONObject) body);
428                return new Response(request, connection, graphObject, isFromCache);
429            } else if (body instanceof JSONArray) {
430                GraphObjectList<GraphObject> graphObjectList = GraphObject.Factory.createList(
431                        (JSONArray) body, GraphObject.class);
432                return new Response(request, connection, graphObjectList, isFromCache);
433            }
434            // We didn't get a body we understand how to handle, so pretend we got nothing.
435            object = JSONObject.NULL;
436        }
437 
438        if (object == JSONObject.NULL) {
439            return new Response(request, connection, (GraphObject)null, isFromCache);
440        } else {
441            throw new FacebookException("Got unexpected object type in response, class: "
442                    + object.getClass().getSimpleName());
443        }
444    }
