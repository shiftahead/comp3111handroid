602    public static List<LatLng> decode(final String encodedPath) {
603        int len = encodedPath.length();
604 
605        // For speed we preallocate to an upper bound on the final length, then
606        // truncate the array before returning.
607        final List<LatLng> path = new ArrayList<LatLng>();
608        int index = 0;
609        int lat = 0;
610        int lng = 0;
611 
612        while (index < len) {
613            int result = 1;
614            int shift = 0;
615            int b;
616            do {
617                b = encodedPath.charAt(index++) - 63 - 1;
618                result += b << shift;
619                shift += 5;
620            } while (b >= 0x1f);
621            lat += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
622 
623            result = 1;
624            shift = 0;
625            do {
626                b = encodedPath.charAt(index++) - 63 - 1;
627                result += b << shift;
628                shift += 5;
629            } while (b >= 0x1f);
630            lng += (result & 1) != 0 ? ~(result >> 1) : (result >> 1);
631 
632            path.add(new LatLng(lat * 1e-5, lng * 1e-5));
633        }
634 
635        return path;
